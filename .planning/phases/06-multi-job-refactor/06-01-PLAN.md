```typescript
export const jobEntrySchema = z.object({
  openingId: z
    .string()
    .min(1)
    .refine((v) => !v.startsWith('REPLACE_WITH'), {
      message: 'openingId must be set to a real BambooHR job opening ID',
    }),
  stages: z.object({
    intake: z.string().min(1),
    pass: z.string().min(1),
    fail: z.string().min(1),
  }),
  hardRules: z
    .object({
      maxSalary: maxSalaryRuleSchema.optional(),
      requiredFields: requiredFieldsRuleSchema.optional(),
      requiredBoolean: z.array(requiredBooleanRuleSchema).optional(),
      requiredKeyword: z.array(requiredKeywordRuleSchema).optional(),
    })
    .refine(
      (rules) =>
        rules.maxSalary !== undefined ||
        rules.requiredFields !== undefined ||
        (rules.requiredBoolean !== undefined && rules.requiredBoolean.length > 0) ||
        (rules.requiredKeyword !== undefined && rules.requiredKeyword.length > 0),
      { message: 'hardRules must contain at least one rule' },
    ),
  fieldMap: z.record(z.string(), z.string()),
  softRules: softRulesSchema,
});

export type JobConfig = z.infer<typeof jobEntrySchema>;

export const configSchema = z.object({
  jobs: z.array(jobEntrySchema).min(1),
});

export type Config = z.infer<typeof configSchema>;
```

<threat_model>
| Threat | Severity | Mitigation |
|--------|----------|------------|
| Malformed `jobs[]` entry passes Zod (e.g., missing `openingId`) and silently runs against wrong job | HIGH | `jobEntrySchema` validates each entry individually with `.min(1)` on all string fields; `configSchema` uses `.min(1)` on the array so zero-entry configs are caught at load time |
| Legacy normalization block misidentifies a non-job object as a legacy config (e.g., a YAML with a `job:` key for unrelated purposes) | MEDIUM | Normalization only triggers when `'job' in raw && !('jobs' in raw)` — the subsequent `configSchema.safeParse` will reject the normalized shape if it does not match `jobEntrySchema`, producing a clear `ConfigError` |
| `openingId` placeholder (`REPLACE_WITH_*`) slips through to a live run and screens real BambooHR candidates under wrong opening | HIGH | `jobEntrySchema` `.refine()` rejects any `openingId` starting with `REPLACE_WITH`; load-time failure prevents any API call |
| `hardRules` object with zero rules (empty `{}`) passes silently and every candidate passes hard rules | HIGH | `.refine()` on `hardRules` inside `jobEntrySchema` requires at least one of `maxSalary`, `requiredFields`, `requiredBoolean`, `requiredKeyword` to be present |
| Lock file race (SAFE-03 deferred) — two cron invocations overlap and double-process candidates | MEDIUM | Accepted — SAFE-03 deliberately deferred from Phase 6 (see CONTEXT.md `<deferred>`). Mitigated operationally by cron interval > expected run time |
</threat_model>
