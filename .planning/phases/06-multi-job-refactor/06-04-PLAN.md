```typescript
// src/screener/multi-job-orchestrator.ts
// Phase 6, D-04: Top-level multi-job batch orchestrator.
// Loops over config.jobs[], instantiates one JobRunner per job, runs sequentially,
// catches per-job failures (D-09), aggregates results, emits D-08 JSON summary.
//
// MULTI-01: All jobs processed in a single container run.
// MULTI-02: Per-job failure isolation — StageValidationError or any throw caught here.
// MULTI-03: Final stdout JSON includes per-job counts and aggregate totals.
// D-10: run() always resolves — never rejects — process.exit handled in index.ts only.
import type { IBambooHRClient } from '../interfaces/IBambooHRClient.js';
import type { ISoftEvaluator } from '../interfaces/ISoftEvaluator.js';
import type { ILogger } from '../interfaces/ILogger.js';
import type { Config } from '../config/schema.js';
import { CandidateProcessor } from '../pipeline/candidate-processor.js';
import { LiveModeWriter } from '../pipeline/live-mode-writer.js';
import { JobRunner } from './job-runner.js';
import type { JobResult, SuccessJobResult } from './job-runner.js';

export class MultiJobOrchestrator {
  constructor(
    private readonly bambooHrClient: IBambooHRClient,
    private readonly softEvaluator: ISoftEvaluator,
    private readonly logger: ILogger,
    private readonly liveWriter: LiveModeWriter,
    private readonly config: Config,
    private readonly dryRun: boolean,
  ) {}

  async run(): Promise<void> {
    const jobResults: JobResult[] = [];

    for (const job of this.config.jobs) {
      try {
        const candidateProcessor = new CandidateProcessor(
          this.bambooHrClient,
          this.softEvaluator,
          this.logger,
          this.liveWriter,
          job,
          this.dryRun,
        );
        const runner = new JobRunner(
          this.bambooHrClient,
          candidateProcessor,
          this.logger,
          job,
          this.dryRun,
        );
        const result = await runner.run();
        jobResults.push(result);
      } catch (err) {
        // D-09: per-job error isolation — any throw (StageValidationError, network error, etc.)
        // is caught here; remaining jobs continue processing.
        const errorReason = err instanceof Error ? err.message : String(err);
        jobResults.push({ openingId: job.openingId, error: true, errorReason });
      }
    }

    // D-08/D-09: aggregate totals — exclude error jobs from counts
    const successJobs = jobResults.filter(
      (r): r is SuccessJobResult => !('error' in r),
    );

    const totals = {
      processed:   successJobs.reduce((sum, r) => sum + r.processed, 0),
      pass:        successJobs.reduce((sum, r) => sum + r.pass, 0),
      fail:        successJobs.reduce((sum, r) => sum + r.fail, 0),
      needsReview: successJobs.reduce((sum, r) => sum + r.needsReview, 0),
      errors:      successJobs.reduce((sum, r) => sum + r.errors, 0),
    };

    // stderr: human-readable summary
    console.error(
      `[main] Done. jobs=${jobResults.length} totals=${JSON.stringify(totals)}`,
    );
    // stdout: machine-readable JSON (INFRA-03, MULTI-03, D-08)
    console.log(JSON.stringify({ jobs: jobResults, totals }));
    // D-10: run() resolves here — never rejects; index.ts main().catch handles unexpected throws
  }
}
```

Key constraints:
- `process.exit` is NOT called here
- `console.log` is called EXACTLY ONCE at the end with the D-08 JSON shape
- The catch block catches ALL errors (not just StageValidationError) — this ensures job-level isolation for any unexpected throw
- `CandidateProcessor` is constructed fresh per-job inside the loop — no shared instance across jobs

```typescript
// src/__tests__/MultiJobOrchestrator.test.ts
// Integration tests for MultiJobOrchestrator.
// Covers: N-job iteration, per-job failure isolation (MULTI-02), aggregate count (MULTI-03),
// D-08 JSON summary shape, D-09 error job representation, D-10 no-reject guarantee.
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';
import { MultiJobOrchestrator } from '../screener/multi-job-orchestrator.js';
import { JobRunner } from '../screener/job-runner.js';
import { StageValidationError } from '../bamboohr/errors.js';
import type { IBambooHRClient } from '../interfaces/IBambooHRClient.js';
import type { ISoftEvaluator } from '../interfaces/ISoftEvaluator.js';
import type { ILogger } from '../interfaces/ILogger.js';
import type { Config } from '../config/schema.js';
import { LiveModeWriter } from '../pipeline/live-mode-writer.js';

function makeConfig(jobCount = 2): Config {
  const jobs = Array.from({ length: jobCount }, (_, i) => ({
    openingId: `job-${i + 1}`,
    stages: { intake: 'New', pass: 'Pass', fail: 'Fail' },
    hardRules: { maxSalary: { value: 100000, label: 'Salary ceiling' } },
    fieldMap: { salary: 'desiredSalary' },
    softRules: undefined,
  }));
  return { jobs } as Config;
}

function makeBambooMock(): IBambooHRClient {
  return {
    get: vi.fn(),
    postComment: vi.fn(),
    moveStage: vi.fn(),
    validateStages: vi.fn().mockResolvedValue(new Map([['New', 1], ['Pass', 2], ['Fail', 3]])),
    fetchApplicationDetails: vi.fn(),
    downloadPdf: vi.fn(),
    fetchCandidates: vi.fn().mockResolvedValue([]),
  } as unknown as IBambooHRClient;
}

function makeLoggerMock(): ILogger {
  return { logDecision: vi.fn(), logEvaluation: vi.fn() };
}

function makeLiveWriter(bambooMock: IBambooHRClient): LiveModeWriter {
  return new LiveModeWriter(bambooMock);
}

function makeOrchestrator(config: Config): {
  orchestrator: MultiJobOrchestrator;
  bambooMock: IBambooHRClient;
  loggerMock: ILogger;
} {
  const bambooMock = makeBambooMock();
  const softMock: ISoftEvaluator = { evaluate: vi.fn() };
  const loggerMock = makeLoggerMock();
  const liveWriter = makeLiveWriter(bambooMock);
  const orchestrator = new MultiJobOrchestrator(
    bambooMock,
    softMock,
    loggerMock,
    liveWriter,
    config,
    true, // dryRun
  );
  return { orchestrator, bambooMock, loggerMock };
}

describe('MultiJobOrchestrator.run', () => {
  let stdoutSpy: ReturnType<typeof vi.spyOn>;
  let stderrSpy: ReturnType<typeof vi.spyOn>;

  beforeEach(() => {
    stdoutSpy = vi.spyOn(console, 'log').mockImplementation(() => {});
    stderrSpy = vi.spyOn(console, 'error').mockImplementation(() => {});
  });
  afterEach(() => {
    stdoutSpy.mockRestore();
    stderrSpy.mockRestore();
    vi.restoreAllMocks();
  });

  it('processes all jobs and emits D-08 aggregate JSON when both succeed', async () => {
    const config = makeConfig(2);
    const { orchestrator } = makeOrchestrator(config);

    let callCount = 0;
    vi.spyOn(JobRunner.prototype, 'run').mockImplementation(async function (this: JobRunner) {
      callCount++;
      if (callCount === 1) {
        return { openingId: 'job-1', processed: 3, pass: 2, fail: 1, needsReview: 0, errors: 0 };
      }
      return { openingId: 'job-2', processed: 2, pass: 1, fail: 1, needsReview: 0, errors: 0 };
    });

    await orchestrator.run();

    expect(stdoutSpy).toHaveBeenCalledWith(
      JSON.stringify({
        jobs: [
          { openingId: 'job-1', processed: 3, pass: 2, fail: 1, needsReview: 0, errors: 0 },
          { openingId: 'job-2', processed: 2, pass: 1, fail: 1, needsReview: 0, errors: 0 },
        ],
        totals: { processed: 5, pass: 3, fail: 2, needsReview: 0, errors: 0 },
      }),
    );
  });

  it('isolates per-job failure — job 2 StageValidationError does not abort job 1 (MULTI-02)', async () => {
    const config = makeConfig(2);
    const { orchestrator } = makeOrchestrator(config);

    let callCount = 0;
    vi.spyOn(JobRunner.prototype, 'run').mockImplementation(async () => {
      callCount++;
      if (callCount === 1) {
        return { openingId: 'job-1', processed: 5, pass: 3, fail: 1, needsReview: 1, errors: 0 };
      }
      throw new StageValidationError('Stage "Applied" not found in BambooHR');
    });

    await orchestrator.run();

    expect(stdoutSpy).toHaveBeenCalledWith(
      JSON.stringify({
        jobs: [
          { openingId: 'job-1', processed: 5, pass: 3, fail: 1, needsReview: 1, errors: 0 },
          { openingId: 'job-2', error: true, errorReason: 'Stage "Applied" not found in BambooHR' },
        ],
        totals: { processed: 5, pass: 3, fail: 1, needsReview: 1, errors: 0 },
      }),
    );
  });

  it('emits jobs array with all error entries and zero totals when all jobs fail (D-10: resolves)', async () => {
    const config = makeConfig(2);
    const { orchestrator } = makeOrchestrator(config);

    vi.spyOn(JobRunner.prototype, 'run').mockRejectedValue(
      new StageValidationError('Boom'),
    );

    // D-10: run() must resolve, not reject
    await expect(orchestrator.run()).resolves.toBeUndefined();

    expect(stdoutSpy).toHaveBeenCalledWith(
      JSON.stringify({
        jobs: [
          { openingId: 'job-1', error: true, errorReason: 'Boom' },
          { openingId: 'job-2', error: true, errorReason: 'Boom' },
        ],
        totals: { processed: 0, pass: 0, fail: 0, needsReview: 0, errors: 0 },
      }),
    );
  });

  it('instantiates one JobRunner per job (MULTI-01)', async () => {
    const config = makeConfig(3);
    const { orchestrator } = makeOrchestrator(config);

    const runSpy = vi.spyOn(JobRunner.prototype, 'run').mockResolvedValue({
      openingId: 'job-1', processed: 1, pass: 1, fail: 0, needsReview: 0, errors: 0,
    });

    await orchestrator.run();

    expect(runSpy).toHaveBeenCalledTimes(3);
  });

  it('emits stdout exactly once with the D-08 JSON shape (MULTI-03)', async () => {
    const config = makeConfig(1);
    const { orchestrator } = makeOrchestrator(config);

    vi.spyOn(JobRunner.prototype, 'run').mockResolvedValue({
      openingId: 'job-1', processed: 2, pass: 1, fail: 1, needsReview: 0, errors: 0,

    await orchestrator.run();

    expect(stdoutSpy).toHaveBeenCalledTimes(1);
    const output = JSON.parse((stdoutSpy.mock.calls[0] as [string])[0]);
    expect(output).toHaveProperty('jobs');
    expect(output).toHaveProperty('totals');
    expect(Array.isArray(output.jobs)).toBe(true);
  });
});
```
